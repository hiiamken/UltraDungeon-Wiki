# Room Files

Each dungeon has its own configuration file at `plugins/UltraDungeon/room/<name>.yml`. These are auto-generated by the setup wizard.

## Full Example

```yaml
# Basic Info
Prefix: "&c&lFire Dungeon"
Player: 4                    # Max players
Time: 10                     # Time limit (minutes)
Spawn:
  ==: org.bukkit.Location
  world: world
  x: 100.0
  y: 65.0
  z: 200.0

# Border
BorderPos1: {==: org.bukkit.Location, world: world, x: 80, y: 50, z: 180}
BorderPos2: {==: org.bukkit.Location, world: world, x: 120, y: 80, z: 220}

# Access Control
Access:
  TicketEnabled: true
  TicketCost: 1
  PermissionEnabled: false
  PermissionNode: "ultradungeon.access.fire_dungeon"

# HP Scaling
HealthScale:
  Enabled: true
  1: 1.0
  2: 1.5
  3: 2.0
  4: 3.0

# Waves
Mob1:
  group_1:
    Type: ZOMBIE
    Amount: 5
    Time: 2
    SpawnStyle: NORMAL
Mob2:
  group_1:
    Type: SKELETON
    Amount: 8
    Time: 2

# Boss
Boss:
  Type: mythic:FireDragon
  Amount: 1
  Time: 5

# Rewards
Reward:
  reward_1:
    Chance: 100
    Command: "eco give %player% 500"
  reward_2:
    Chance: 50
    Item: {==: org.bukkit.inventory.ItemStack, ...}
RewardAmount: 1

# Vault Integration
MoneyReward: 1000.0
EntryFee: 100.0

# MMOCore Integration
MMOCoreExpReward: 200.0

# MMOItems Integration
MMOItemRewards:
  sword_1:
    Type: SWORD
    Id: FLAME_BLADE
    Amount: 1

# Default Rewards (guaranteed)
DefaultRewards:
  enabled: true
  rewards:
    coins:
      type: command
      command: "eco give %player% 100"
      display-name: "&6100 Coins"

# Card Flip Rewards
CardFlip:
  enabled: false

# Checkpoints
Checkpoints:
  cp_1:
    location: {==: org.bukkit.Location, ...}
    trigger: AUTO
    trigger-stage: 3

# Victory Commands
VictoryCommands:
  - "broadcast %player% cleared Fire Dungeon!"

# Progression Requirements
Requirements:
  completed-dungeons:
    - "tutorial"
  min-level: 5
```
